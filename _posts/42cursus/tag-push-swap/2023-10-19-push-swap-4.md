### (1) void  swap_top(t_deque *a)

~~~c
void	swap_top(t_deque *a)
{
	int		tmp;
	int		idx1;
	int		idx2;

	idx1 = a->head;
	if (idx1 == a->capacity - 1)
		idx2 = 0;
	else
		idx2 = idx1 + 1;
	tmp = a->arr[idx1];
	a->arr[idx1] = a->arr[idx2];
	a->arr[idx2] = tmp;
}
~~~
