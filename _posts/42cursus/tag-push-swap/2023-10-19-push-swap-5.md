### void	resize(t_deque *d, int new_cap)

~~~c
void	resize(t_deque *d, int new_cap)
{
	int		*new_arr;
	int		idx;
	int		i;

	new_arr = (int *) malloc(sizeof(int) * new_cap);
	idx = 0;
	i = d->head - 1;
	if (d->head < d->tail)
		while (++i < d->tail)
			new_arr[idx++] = d->arr[i];
	else
	{
		while (++i < d->capacity)
			new_arr[idx++] = d->arr[i];
		i = -1;
		while (++i < d->tail)
			new_arr[idx++] = d->arr[i];
	}
	free(d->arr);
	d->arr = new_arr;
	d->head = 0;
	d->tail = idx;
	d->capacity = new_cap;
}
~~~
